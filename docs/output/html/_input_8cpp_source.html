43void Input::Update() {
44 // Logika edge-trigger:
45 // - previous <- current
46 // - current <- odczyt z urzadzenia
47 // Dzieki temu mozna wykryc przejscie 0->1 (JustPressed).
48 previousKeyStates = currentKeyStates;
49 for (auto& pair : currentKeyStates) {
50 pair.second = sf::Keyboard::isKeyPressed(pair.first);
51 }
52
53 previousMouseStates = currentMouseStates;
54 for (auto& pair : currentMouseStates) {
55 // Sprawdzamy, czy przycisk myszy jest aktualnie wciśnięty.
56 pair.second = sf::Mouse::isButtonPressed(pair.first);
57 }
58}

63bool Input::IsActionJustPressed(const std::string& actionName) {
64 auto itKey = actionBindings.find(actionName);
65 if (itKey != actionBindings.end()) {
66 KeyCode key = itKey->second;
67 return currentKeyStates[key] && !previousKeyStates[key];
68 }
69
70 auto itMouse = mouseActionBindings.find(actionName);
71 if (itMouse != mouseActionBindings.end()) {
72 sf::Mouse::Button btn = itMouse->second;
73 return currentMouseStates[btn] && !previousMouseStates[btn];
74 }
75
76 return false;
77}
