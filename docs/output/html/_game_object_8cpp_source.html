74Point2D GameObject::GetWorldPosition() const {
75 if (parent) {
76 // Kluczowa transformacja hierarchiczna (scene graph):
77 // p_world = p_parent_world + R(parent_angle) * p_local
78 // gdzie R(theta) to macierz rotacji 2D.
79 return parent->GetWorldPosition() + position.Rotate(parent->GetWorldAngle());
80 }
81 // Korzen drzewa: pozycja lokalna jest rowna globalnej.
