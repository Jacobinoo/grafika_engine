13void Engine::Init(const EngineConfig& config) {
14 // 1) Ustawienia domyslne okna.
15 unsigned int windowStyle = sf::Style::Default;
16 sf::State windowState = sf::State::Windowed;
17
18 // 2) Parametry aktualnego trybu pulpitu.
19 sf::VideoMode desktop = sf::VideoMode::getDesktopMode();
20
21 // Tworzymy tryb bazowy z zadanej szerokosci/wysokosci.
22 sf::VideoMode videoMode({(unsigned int)config.width, (unsigned int)config.height}, desktop.bitsPerPixel);
23
24 // 3) Obsluga trybu pelnoekranowego.
25 if (config.windowMode == WindowMode::FULLSCREEN) {
26 windowState = sf::State::Fullscreen;
27 bool isModeSupported = false;
28
29 // Sprawdzamy, czy zadany tryb jest wspierany przez monitor.
30 std::vector<sf::VideoMode> modes = sf::VideoMode::getFullscreenModes();
31
32 for (const auto& mode : modes) {
33 if (mode.size.x == config.width && mode.size.y == config.height) {
34 isModeSupported = true;
35 videoMode = mode;
36 break;
37 }
38 }
39
40 // Fallback: przechodzimy do natywnego trybu pulpitu.
41 if (!isModeSupported) {
42 std::cout << "[Engine] OSTRZEZENIE: Rozdzielczosc " << config.width << "x" << config.height
43 << " nie jest obslugiwana w Fullscreen na tym monitorze!" << std::endl;
44
45 std::cout << "[Engine] Wymuszam natywna rozdzielczosc pulpitu: "
46 << desktop.size.x << "x" << desktop.size.y << std::endl;
47
48 videoMode = desktop;
49 }
50 }
51 else if (config.windowMode == WindowMode::BORDERLESS) {
52 windowStyle = sf::Style::None;
53 }
54
55 // 4) Tworzenie okna.
56 window = std::make_unique<sf::RenderWindow>(videoMode, config.title, windowStyle, windowState);
57
58 // 5) Konfiguracja FPS.
59 window->setVerticalSyncEnabled(false);
60 window->setFramerateLimit(config.framerateLimit);
61 targetFramerate = config.framerateLimit;
62
63 if(config.framerateLimit == 0) {
64 window->setFramerateLimit(60);
65 targetFramerate = 60;
66 }
67
68 // 6) Inicjalizacja renderera programowego.
69 renderer = std::make_unique<Renderer>(videoMode.size.x, videoMode.size.y);
70
71 // Rezerwacja redukuje realokacje i fragmentacje pamieci podczas tworzenia obiektow.
72 gameObjects.reserve(1000);
73 objectQueue.reserve(100);
74
75 isRunning = true;
76}

81void Engine::Run(IGame* game) {
82 if(game) game->Start();
83
84 while (window->isOpen() && isRunning) {
85 // 1) Obsluga zdarzen systemowych.
86 while (const std::optional event = window->pollEvent()) {
87 if (event->is<sf::Event::Closed>()) {
88 Quit();
89 }
90 }
91
92 // 2) Snapshot wejscia dla tej klatki.
93 Input::Update();
94
95 // 3) Kod gry uzytkownika.
96 if (game) game->Update();
97
98 // 4) Aktualizacja aktywnych obiektow.
99 for (auto& obj : gameObjects) if (obj->isActive) obj->Update();
100
101 // Kolejka obiektow zapobiega invalidacji iteratorow podczas petli Update().
102 for (auto& newObj : objectQueue) {
103 gameObjects.push_back(std::move(newObj));
104 }
105 objectQueue.clear();
106
107 // Remove-erase idiom: liniowe usuniecie obiektow oznaczonych jako nieaktywne.
108 gameObjects.erase(
109 std::remove_if(gameObjects.begin(), gameObjects.end(),
110 [](const std::unique_ptr<GameObject>& obj) { return !obj->isActive; }),
111 gameObjects.end()
112 );
113
114 // 5) Render klatki.
115 window->clear(currentBackgroundColor);
116 renderer->Clear(currentBackgroundColor);
117
118 for (auto& obj : gameObjects) {
119 obj->Render(*renderer);
120 }
121
122 renderer->RenderToWindow(*window);
123 window->display();
124 }
125
126 // Sprzatanie po zakonczeniu petli.
127 ClearScene();
128 if (window) window.reset();
129 if (renderer) renderer.reset();
130}
