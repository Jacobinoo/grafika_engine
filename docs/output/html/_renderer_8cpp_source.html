51void Renderer::Draw(const LineSegment& line, sf::Color color) {
52 int x0 = std::round(line.p1.x);
53 int y0 = std::round(line.p1.y);
54 int x1 = std::round(line.p2.x);
55 int y1 = std::round(line.p2.y);
56
57 int dx = std::abs(x1 - x0), sx = x0 < x1 ? 1 : -1;
58 int dy = -std::abs(y1 - y0), sy = y0 < y1 ? 1 : -1;
59 int err = dx + dy, e2;
60
61 // Bresenham wykorzystuje blad calkowitoliczbowy zamiast kosztownych operacji float.
62 while (true) {
63 PutPixel(x0, y0, color);
64 if (x0 == x1 && y0 == y1) break;
65 e2 = 2 * err;
66 if (e2 >= dy) { err += dy; x0 += sx; }
67 if (e2 <= dx) { err += dx; y0 += sy; }
68 }
69}

102void Renderer::Draw(const Circle& circle, sf::Color color) {
103 int cx = std::round(circle.center.x);
104 int cy = std::round(circle.center.y);
105 int x = 0;
106 int y = std::round(circle.radius);
107 int d = 3 - 2 * y;
108
109 // Symetria osmiu oktantow pozwala obliczac tylko 1/8 okregu.
110 auto drawOctants = [&](int xc, int yc, int x, int y) {
111 PutPixel(xc + x, yc + y, color); PutPixel(xc - x, yc + y, color);
112 PutPixel(xc + x, yc - y, color); PutPixel(xc - x, yc - y, color);
113 PutPixel(xc + y, yc + x, color); PutPixel(xc - y, yc + x, color);
114 PutPixel(xc + y, yc - x, color); PutPixel(xc - y, yc - x, color);
115 };
116
117 while (y >= x) {
118 drawOctants(cx, cy, x, y);
119 x++;
120 if (d > 0) { y--; d = d + 4 * (x - y) + 10; }
121 else { d = d + 4 * x + 6; }
122 }
123}

152void Renderer::Fill(const Triangle& tri, sf::Color color) {
153 int minX = std::max(0, (int)std::floor(std::min({tri.p1.x, tri.p2.x, tri.p3.x})));
154 int maxX = std::min((int)width - 1, (int)std::ceil(std::max({tri.p1.x, tri.p2.x, tri.p3.x})));
155 int minY = std::max(0, (int)std::floor(std::min({tri.p1.y, tri.p2.y, tri.p3.y})));
156 int maxY = std::min((int)height - 1, (int)std::ceil(std::max({tri.p1.y, tri.p2.y, tri.p3.y})));
157
158 // Dla kazdego piksela z AABB sprawdzamy czy lezy po tej samej stronie kazdej krawedzi.
159 for (int y = minY; y <= maxY; ++y) {
160 for (int x = minX; x <= maxX; ++x) {
161 Point2D pt((float)x, (float)y);
162 bool b1 = EdgeSign(pt, tri.p1, tri.p2) < 0.0f;
163 bool b2 = EdgeSign(pt, tri.p2, tri.p3) < 0.0f;
164 bool b3 = EdgeSign(pt, tri.p3, tri.p1) < 0.0f;
165
166 if ((b1 == b2) && (b2 == b3)) {
167 PutPixel(x, y, color);
168 }
169 }
170 }

176void Renderer::Fill(const Polygon& poly, sf::Color color) {
177 if (poly.vertices.size() < 3) return;
178
179 float minY = poly.vertices[0].y;
180 float maxY = poly.vertices[0].y;
181 for (const auto& v : poly.vertices) {
182 if (v.y < minY) minY = v.y;
183 if (v.y > maxY) maxY = v.y;
184 }
185
186 // Dla kazdej linii poziomej Y znajdujemy przecięcia z krawedziami wielokata,
187 // sortujemy je i wypelniamy przedzialy parami.
188 for (int y = (int)minY; y <= (int)maxY; ++y) {
189 std::vector<int> intersections;
190
191 for (size_t i = 0; i < poly.vertices.size(); ++i) {
192 Point2D p1 = poly.vertices[i];
193 Point2D p2 = poly.vertices[(i + 1) % poly.vertices.size()];
194
195 if (p1.y == p2.y) continue;
196
197 if (y >= std::min(p1.y, p2.y) && y < std::max(p1.y, p2.y)) {
198 int intersectX = (int)(p1.x + (y - p1.y) * (p2.x - p1.x) / (p2.y - p1.y));
199 intersections.push_back(intersectX);
200 }
201 }
202
203 std::sort(intersections.begin(), intersections.end());
204
205 for (size_t i = 0; i + 1 < intersections.size(); i += 2) {
206 for (int x = intersections[i]; x <= intersections[i + 1]; ++x) {
207 PutPixel(x, y, color);
208 }
209 }
210 }

216void Renderer::Fill(const Circle& circle, sf::Color color) {
217 int cx = std::round(circle.center.x);
218 int cy = std::round(circle.center.y);
219 int radius = std::round(circle.radius);
220
221 int x = 0;
222 int y = radius;
223 int d = 3 - 2 * radius;
224
225 // Wypelnienie realizujemy przez poziome odcinki pomiedzy parami punktow symetrycznych.
226 auto drawHorizontalLine = [&](int x1, int x2, int y_row) {
227 if (y_row < 0 || y_row >= (int)height) return;
228
229 int startX = std::max(0, std::min(x1, x2));
230 int endX = std::min((int)width - 1, std::max(x1, x2));
231
232 for (int px = startX; px <= endX; ++px) {
233 PutPixel(px, y_row, color);
234 }
235 };
236
237 while (y >= x) {
238 // Cztery scanline odpowiadaja ośmiopunktowej symetrii okregu.
239 drawHorizontalLine(cx - x, cx + x, cy + y);
240 drawHorizontalLine(cx - x, cx + x, cy - y);
241 drawHorizontalLine(cx - y, cx + y, cy + x);
242 drawHorizontalLine(cx - y, cx + y, cy - x);
243
244 x++;
245 if (d > 0) {
246 y--;
247 d = d + 4 * (x - y) + 10;
248 } else {
249 d = d + 4 * x + 6;
250 }
251 }

257void Renderer::DrawBitmapText(const std::string& text, int startX, int startY, const sf::Image& fontImage, float scale) {
258 int cellW = 64;
259 int cellH = 64;
260
261 int scaledW = cellW * scale;
262 int scaledH = cellH * scale;
263
264 int cursorX = startX;
265
266 for (char c : text) {
267 // Kod ASCII indeksuje pozycje znaku na siatce 16x16.
268 int ascii = (unsigned char)c;
269
270 int col = ascii % 16;
271 int row = ascii / 16;
272
273 int texStartX = col * cellW;
274 int texStartY = row * cellH;
275
276 for (int sy = 0; sy < scaledH; ++sy) {
277 for (int sx = 0; sx < scaledW; ++sx) {
278
279 // Skalowanie "wsteczne": piksel docelowy mapujemy na zrodlo.
280 int texX = texStartX + (sx / scale);
281 int texY = texStartY + (sy / scale);
282
283 sf::Vector2u vec2 = sf::Vector2u{(uint) texX, (uint) texY};
284
285 sf::Color pixelColor = fontImage.getPixel(vec2);
286
287 if (pixelColor.a > 128 && pixelColor != sf::Color::Black) {
288 PutPixel(cursorX + sx, startY + sy, pixelColor);
289 }
290 }
291 }
292 cursorX += scaledW;
293 }
