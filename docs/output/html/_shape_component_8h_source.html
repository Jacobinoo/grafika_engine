65 if (!gameObject) return;
66
67 Point2D pos = gameObject->GetWorldPosition();
68 float rot = gameObject->GetWorldAngle();
69
70 // Dla kazdego typu figury wykonujemy ten sam pipeline:
71 // 1) transformacja lokalna -> swiat,
72 // 2) wybor Draw/Fill.
73 std::visit([&](auto&& arg) {
74 using T = std::decay_t<decltype(arg)>;
75
76 if constexpr (std::is_same_v<T, Circle>) {
77 // Kolo: obracamy tylko srodek lokalny i przesuwamy do pozycji obiektu.
78 Point2D worldCenter = arg.center.Rotate(rot) + pos;
79 Circle worldCircle(worldCenter, arg.radius);
80
81 if (isFilled) renderer.Fill(worldCircle, color);
82 else renderer.Draw(worldCircle, color);
83 }
84 else if constexpr (std::is_same_v<T, Polygon>) {
85 std::vector<Point2D> worldVerts;
86 for (const auto& v : arg.vertices) {
87 worldVerts.push_back(v.Rotate(rot) + pos);
88 }
89 Polygon worldPoly(worldVerts);
90
91 if (isFilled) renderer.Fill(worldPoly, color);
92 else renderer.Draw(worldPoly, color);
93 }
94 else if constexpr (std::is_same_v<T, Rectangle>) {
95 // Po rotacji prostokat osiowy staje sie czworokatem.
96 // Dlatego konwertujemy go do wielokata i rysujemy ogolnym algorytmem.
97 std::vector<Point2D> verts = {
98 arg.topLeft,
99 Point2D(arg.topLeft.x + arg.width, arg.topLeft.y),
100 Point2D(arg.topLeft.x + arg.width, arg.topLeft.y + arg.height),
101 Point2D(arg.topLeft.x, arg.topLeft.y + arg.height)
102 };
103
104 std::vector<Point2D> worldVerts;
105 for (const auto& v : verts) {
106 worldVerts.push_back(v.Rotate(rot) + pos);
107 }
108
109 Polygon worldPoly(worldVerts);
110 if (isFilled) renderer.Fill(worldPoly, color);
111 else renderer.Draw(worldPoly, color);
112 }
113 else if constexpr (std::is_same_v<T, Triangle>) {
114 // Trojkat: transformujemy trzy lokalne wierzcholki.
115 Point2D p1 = arg.p1.Rotate(rot) + pos;
116 Point2D p2 = arg.p2.Rotate(rot) + pos;
117 Point2D p3 = arg.p3.Rotate(rot) + pos;
118
119 Triangle worldTri(p1, p2, p3);
120
121 if (isFilled) renderer.Fill(worldTri, color);
122 else renderer.Draw(worldTri, color);
123 }
124 else if constexpr (std::is_same_v<T, LineSegment>) {
125 // Odcinek: transformujemy dwa konce.
126 Point2D p1 = arg.p1.Rotate(rot) + pos;
127 Point2D p2 = arg.p2.Rotate(rot) + pos;
128
129 LineSegment worldLine(p1, p2);
130
131 // Odcinek jest figura 1D, wiec wypelnienie nie ma sensu.
132 renderer.Draw(worldLine, color);
133 }
134
135 }, shape);
136 }
